Example PV YAML:

apiVersion: v1
kind: PersistentVolume
metadata:
  name: {{ template "nginx-static.fullname" . }}-volume
spec:
  capacity:
    storage: 100Mi
  accessModes:
    - ReadOnlyMany
  persistentVolumeReclaimPolicy: Recycle
  claimRef:
    namespace: default
    name: {{ template "nginx-static.fullname" . }}-html-pvc
  hostPath:
    path: "/var/{{ .Release.Name }}/{{ template "nginx-static.name" . }}/html"
